49  static bool add_process(std::list< process<model_t,N_in,N_out> >& procs,const std::string& str)
50  {
51  vector<std::string,N_in+N_out>namevec;
52  if(!decompose_process<N_in,N_out>(str,namevec))
53  {
54  return false;
55  }
56 
57  /* Constructing vectors of corresponding incoming particles: */
58 
59  std::vector<vector<const particle_type*,N_in> >Istates(1);
60  const particle_type* phi;
61  for(typename std::vector<std::string>::size_type i=0;i<N_in;++i)
62  {
63  if(model_wrapper<model_t>::contains_particle(namevec[i]))
64  {
65  phi=model_wrapper<model_t>::get_particle(namevec[i]);
66  for(typename std::vector<vector<const particle_type*,N_in> >::size_type j=0;j<Istates.size();++j)
67  {
68  Istates[j][i]=phi;
69  }
70  }
71  else if(model_wrapper<model_t>::contains_family(namevec[i]))
72  {
73  const particle_family<model_t>* Ijet=model_wrapper<model_t>::get_family(namevec[i]);
74  typename std::vector<vector<const particle_type*,N_in> >::size_type In=Istates.size();
75  Istates.resize(In*Ijet->size());
76  typename std::vector<vector<const particle_type*,N_in> >::size_type Il=0;
77  for(typename particle_family<model_t>::size_type j=0;j<Ijet->size();++j)
78  {
79  for(typename std::vector<vector<const particle_type*,N_in> >::size_type k=0;k<In;++k)
80  {
81  for(typename vector<const particle_type*,N_in>::size_type l=0;l<i;++l)
82  {
83  Istates[Il][l]=Istates[k][l];
84  }
85  Istates[Il][i]=(*Ijet)[j];
86  ++Il;
87  }
88  }
89  }
90  else
91  {
92  log(log_level::warning)<<CAMGEN_STREAMLOC<<"particle or family "<<namevec[i]<<" not found, ignoring process "<<str<<endlog;
93  return false;
94  }
95  }
96 
97  /* Constructing vectors of corresponding outgoing particles: */
98 
99  std::vector<vector<const particle_type*,N_out> >Fstates(1);
100  for(typename std::vector<std::string>::size_type i=0;i<N_out;++i)
101  {
102  if(model_wrapper<model_t>::contains_particle(namevec[N_in+i]))
103  {
104  const particle_type* phi=model_wrapper<model_t>::get_particle(namevec[N_in+i]);
105  for(typename std::vector<vector<const particle_type*,N_out> >::size_type j=0;j<Fstates.size();++j)
106  {
107  Fstates[j][i]=phi;
108  }
109  }
110  else if(model_wrapper<model_t>::contains_family(namevec[N_in+i]))
111  {
112  const particle_family<model_t>* Fjet=model_wrapper<model_t>::get_family(namevec[N_in+i]);
113  typename std::vector<vector<const particle_type*,N_out> >::size_type Fn=Fstates.size();
114  Fstates.resize(Fn*Fjet->size());
115  typename std::vector<vector<const particle_type*,N_out> >::size_type Fl=0;
116 
117  for(typename particle_family<model_t>::size_type j=0;j<Fjet->size();++j)
118  {
119  for(typename std::vector<vector<const particle_type*,N_out> >::size_type k=0;k<Fn;++k)
120  {
121  for(typename vector<const particle_type*,N_out>::size_type l=0;l<i;++l)
122  {
123  Fstates[Fl][l]=Fstates[k][l];
124  }
125  Fstates[Fl][i]=(*Fjet)[j];
126  ++Fl;
127  }
128  }
129  }
130  else
131  {
132  log(log_level::warning)<<CAMGEN_STREAMLOC<<"particle or family "<<namevec[N_in+i]<<" not found, ignoring process "<<str<<endlog;
133  return false;
134  }
135  }
136 
137  /* Adding the constructed processes to the given list: */
138 
139  for(typename std::vector< vector<const particle_type*,N_in> >::size_type i=0;i<Istates.size();++i)
140  {
141  for(typename std::vector< vector<const particle_type*,N_out> >::size_type j=0;j<Fstates.size();++j)
142  {
143  procs.push_back(process<model_t,N_in,N_out>(Istates[i],Fstates[j]));
144  }
145  }
146  return true;
147  }

152  static typename std::list< process<model_t,N_in,N_out> >::iterator insert_process_by_flavour(std::list< process<model_t,N_in,N_out> >& processes,typename std::list< process<model_t,N_in,N_out> >::iterator proc_iter,const vector<size_type,N_external>& fv)
153  {
154  process<model_t,N_in,N_out>p;
155  const particle_type* phi;
156  for(std::size_t i=0;i<N_in;++i)
157  {
158  phi=model_wrapper<model_t>::get_particle(fv[i]);
159  if(phi==NULL)
160  {
161  return processes.end();
162  }
163  p.IS_particles[i]=phi;
164  p.unsorted_pdg_ids[i]=phi->get_pdg_id();
165  }
166  for(std::size_t i=0;i<N_out;++i)
167  {
168  phi=model_wrapper<model_t>::get_particle(fv[N_in+i]);
169  if(phi==NULL)
170  {
171  return processes.end();
172  }
173  p.FS_particles[i]=phi;
174  p.unsorted_pdg_ids[N_in+i]=phi->get_pdg_id();
175  }
176  p.sorted_pdg_ids=p.unsorted_pdg_ids;
177  std::sort(p.sorted_pdg_ids.begin(),p.sorted_pdg_ids.begin()+N_in);
178  std::sort(p.sorted_pdg_ids.begin()+N_in,p.sorted_pdg_ids.end());
179  p.unsorted_flavours=fv;
180  p.sorted_flavours=fv;
181  std::sort(p.sorted_flavours.begin(),p.sorted_flavours.begin()+N_in);
182  std::sort(p.sorted_flavours.begin()+N_in,p.sorted_flavours.end());
183  return processes.insert(proc_iter,p);
184  }
