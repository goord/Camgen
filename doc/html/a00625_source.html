38  /* Initialisation phase, creating the vector of 'jumps' of the iterators
39  * to nonzero entries: */
40 
41  static void initialise()
42  {
43  if(!initialised)
44  {
45  group_type::initialise();
46 
47  /* First, list the nonzero components in the 'jumps' vectors: */
48 
49  value_type f;
50  for(size_type a=0;a<group_t::rank;++a)
51  {
52  for(size_type b=0;b<group_t::rank;++b)
53  {
54  for(size_type c=0;c<group_t::rank;++c)
55  {
56  for(size_type d=0;d<group_t::rank;++d)
57  {
58  f=value_type(0,0);
59  for(size_type e=0;e<group_t::rank;++e)
60  {
61  f+=((group_type::structure_constant(e,a,b))*(group_type::structure_constant(e,c,d)));
62  }
63  if(f!=value_type(0,0))
64  {
65  values.push_back(f);
66  jumps[0].push_back(a*Feynrule_t::sizes[I]);
67  jumps[1].push_back(b*Feynrule_t::sizes[J]);
68  jumps[2].push_back(c*Feynrule_t::sizes[K]);
69  jumps[3].push_back(d*Feynrule_t::sizes[L]);
70  }
71  }
72  }
73  }
74  }
75 
76  /* Secondly, we subtract each jump entry by the previous one to
77  * obtain the offsets: */
78 
79  for(size_type i=0;i<4;++i)
80  {
81  backsets[i]=jumps[i].back();
82  for(int j=size()-1;j!=0;--j)
83  {
84  jumps[i][j]-=jumps[i][j-1];
85  }
86  }
87  initialised=true;
88  }
89  }
90

92

96  }
