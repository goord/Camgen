113  void configure(process_generator_type* procgen,generator_configuration<model_t>* conf)
114  {
115  if(conf!=NULL)
116  {
117  conf->lock_generator(procgen);
118  basic_cuts::clear();
119  conf->configure();
120  }
121 
122  procgen->set_ps_generator(create_momentum_generator(procgen->amplitude));
123  procgen->set_helicity_generator(create_helicity_generator(procgen->amplitude));
124  procgen->set_colour_generator(create_colour_generator(procgen->amplitude));
125 
126  if(Camgen::auto_channel_adapt())
127  {
128  procgen->set_auto_channel_adapt(auto_channel_batch());
129  }
130  if(Camgen::auto_grid_adapt())
131  {
132  procgen->set_auto_grid_adapt(auto_grid_batch());
133  }
134  procgen->max_rejects=max_init_rejects();
135  if(discarded_weight_fraction()>0 and discarded_weight_fraction()<1)
136  {
137  procgen->bin_weights(weight_bins());
138  procgen->reduce_max_weight(discarded_weight_fraction());
139  }
140  for(std::size_t i=1;i<=N_out;++i)
141  {
142  for(std::size_t j=i+1;j<=N_out;++j)
143  {
144  procgen->set_m_min(i,j,basic_cuts::m_min(i,j));
145  }
146  }
147  procgen->insert_scale(conf);
148  procgen->insert_cut(conf);
149  procgen->set_pdf_alpha_s(use_pdf_alpha_s());
150  }
