223  /* Function computing the couplings from the input parameters: */
224 
225  static void refresh_couplings()
226  {
227  Q_e=std::sqrt((value_type)4*pi*alpha);
228 
229  gammaee=std::complex<value_type>(0,Q_e);
230  gammamumu=gammaee;
231  gammatautau=gammaee;
232 
233  gammaseLseL=std::complex<value_type>(0,Q_e);
234  gammaseRseR=gammaseLseL;
235 
236  gammasmuLsmuL=gammaseLseL;
237  gammasmuRsmuR=gammaseLseL;
238 
239  gammastauLstauL=gammaseLseL;
240  gammastauRstauR=gammaseLseL;
241 
242  seLchie=std::complex<value_type>(0,std::sqrt((value_type)2)*Q_e);
243  seRchie=seLchie;
244  seLechi=-seLchie;
245  seRechi=-seLchie;
246 
247  smuLchimu=seLchie;
248  smuRchimu=seLchie;
249  smuLmuchi=-seLchie;
250  smuRmuchi=-seLchie;
251 
252  stauLchitau=seLchie;
253  stauRchitau=seLchie;
254  stauLtauchi=-seLchie;
255  stauRtauchi=-seLchie;
256 
257 
258  seLseLgammagamma=std::complex<value_type>(0,(value_type)2*Q_e*Q_e);
259  seRseRgammagamma=seLseLgammagamma;
260  smuLsmuLgammagamma=seLseLgammagamma;
261  smuRsmuRgammagamma=seLseLgammagamma;
262  stauLstauLgammagamma=seLseLgammagamma;
263  stauRstauRgammagamma=seLseLgammagamma;
264 
265  seLseLseLseL=-seLseLgammagamma;
266  seRseRseRseR=seLseLseLseL;
267  smuLsmuLsmuLsmuL=seLseLseLseL;
268  smuRsmuRsmuRsmuR=seLseLseLseL;
269  stauLstauLstauLstauL=seLseLseLseL;
270  stauRstauRstauRstauR=seLseLseLseL;
271 
272  seLseLseRseR=(value_type)0.5*seLseLgammagamma;
273  smuLsmuLsmuRsmuR=seLseLseRseR;
274  stauLstauLstauRstauR=seLseLseRseR;
275  }
