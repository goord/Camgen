64  /* Generates the invariant masses: */
65 
66  bool generate_s()
67  {
68  if(!refresh_s_bounds())
69  {
70  sweight=(value_type)0;
71  return false;
72  }
73  value_type sqrts=this->m_in();
74  const value_type& sqrts1=this->m_out(0);
75  const value_type& sqrts2=this->m_out(1);
76  if(this->max_s_pairs()>1)
77  {
78  std::size_t n=0;
79  bool q,q1,q2;
80  do
81  {
82  q1=this->channel(0)->generate_s();
83  q2=this->channel(1)->generate_s();
84  q=(q1 and q2 and (sqrts1+sqrts2<=sqrts));
85  ++n;
86  }
87  while(!q and (n<this->max_s_pairs()));
88  if(!q)
89  {
90  log(log_level::warning)<<CAMGEN_STREAMLOC<<"no succesful s-pair generated after "<<n<<" throws--returning weight 0"<<endlog;
91  sweight=(value_type)0;
92  return false;
93  }
94  sweight=(integrate(this->channel(0),this->channel(1),sqrts)*(this->channel(0)->s_weight())*(this->channel(1)->s_weight()));
95  }
96  else
97  {
98  bool q1=this->channel(0)->generate_s();
99  bool q2=this->channel(1)->generate_s();
100  if(!(q1 and q2) or ((sqrts1+sqrts2)>sqrts))
101  {
102  sweight=(value_type)0;
103  return false;
104  }
105  sweight=(this->channel(0)->s_weight())*(this->channel(1)->s_weight());
106  }
107  return (sweight>(value_type)0);
108  }

110  /* Generates momenta with current invariant masses: */
111 
112  bool generate_p()
113  {
114  value_type s=this->s_in();
115  value_type sqrts=this->m_in();
116  value_type s1=this->s_out(0);
117  value_type s2=this->s_out(1);
118  value_type lambda=std::sqrt(Kallen(s,s1,s2));
119  value_type prefactor=(value_type)0.5/sqrts;
120  (this->p_out(0))[0]=prefactor*(s+s1-s2);
121  phat=prefactor*lambda;
122  sphere_generator->generate();
123  for(size_type mu=1;mu<model_t::dimension;++mu)
124  {
125  (this->p_out(0))[mu]=(sphere_generator->object())[mu-1];
126  }
127  boost_from_restframe(this->p_out(0),this->p_in(),sqrts);
128  this->p_out(1)=this->p_in()-this->p_out(0);
129  this->weight()=massless_ps<value_type,2,model_t::dimension>::volume(sqrts)*std::pow(lambda/s,int(model_t::dimension-3))*sweight;
130  return (this->weight()>(value_type)0);
131  }

136  {

138  {
139  sweight=(value_type)0;
140  this->weight()=(value_type)0;
141  return false;
142  }
143  value_type sqrts=this->m_in();
144  if((this->m_out(0)+this->m_out(1)>sqrts) or !this->channel(0)->evaluate_s_weight() or !this->channel(1)->evaluate_s_weight())
145  {
146  sweight=(value_type)0;
147  this->weight()=(value_type)0;
148  return false;
149  }
150  if(this->max_s_pairs()>1)
151  {
152  sweight=integrate(this->channel(0),this->channel(1),sqrts)*(this->channel(0)->s_weight())*(this->channel(1)->s_weight());
153  }
154  else
155  {
156  sweight=(this->channel(0)->s_weight())*(this->channel(1)->s_weight());
157  }
158  value_type s=this->s_in();
159  value_type lambda=std::sqrt(Kallen(s,this->s_out(0),this->s_out(1)));
160  this->weight()=massless_ps<value_type,2,model_t::dimension>::volume(sqrts)*std::pow(lambda/s,int(model_t::dimension-3))*sweight;
161  return (this->weight()>(value_type)0);
162  }
