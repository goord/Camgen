76  bool generate()
77  {
78  if(cached)
79  {
80  this->object()=cache;
81  this->weight()=cached_weight;
82  cached=false;
83  }
84  else
85  {
86  value_type w,x1,x2;
87  do
88  {
89  x1=rn_stream::throw_number(-(value_type)1,(value_type)1);
90  x2=rn_stream::throw_number(-(value_type)1,(value_type)1);
91  w=x1*x1+x2*x2;
92  }
93  while(w>=(value_type)1);
94  value_type y=(sigma==NULL)?(std::sqrt(-(value_type)2*std::log(w)/w)):(std::sqrt(-(value_type)2*std::log(w)/w)*(*sigma));
95  cache=(mu==NULL)?(x1*y):(*mu+x1*y);
96  cached_weight=(sigma==NULL)?(std::pow(w,-x1*x1/w)/factor):(std::abs(*sigma)*std::pow(w,-x1*x1/w)/factor);
97  this->object()=(mu==NULL)?(x2*y):(*mu+x2*y);
98  this->weight()=(sigma==NULL)?(std::pow(w,-x2*x2/w)/factor):(std::abs(*sigma)*std::pow(w,-x2*x2/w)/factor);
99  cached=true;
100  }
101  return true;
102  }
