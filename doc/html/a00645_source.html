23 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
24  * Histogramming facility for Camgen's random number generators. The class *
25  * performs a binning of the data in a range offset*sigma around the mean value, *
26  * over a given number of bins. Then it writes out the bin fillings and the bin *
27  * weights as given by the generator to a data file, together with a gnuplot *
28  * script to visualise the histogram. *

194  void make(bin_size_type bns=100,const value_type& offset_left=3,const value_type& offset_right=3)
195  {
196  N_bins=bns;
197  bins.clear();
198  bins.reserve(N_bins);
199  mu/=(N_points);
200  sig=0;
201  for(data_size_type i=0;i<N_points;++i)
202  {
203  sig+=((data[i].value()-mu)*(data[i].value()-mu));
204  }
205  sig=std::sqrt(sig/N_points);
206 
207  /* Computing minimal and maximal bin and bin size: */
208 
209  x_min=mu-std::abs(offset_left)*sig;
210  x_max=mu+std::abs(offset_right)*sig;
211  bin_size=(x_max-x_min)/N_bins;
212 
213  /* Constructing the bins: */
214 
215  if(weight==NULL)
216  {
217  for(bin_size_type i=0;i<N_bins;++i)
218  {
219  bins.push_back(bin(x_min+i*bin_size));
220  }
221  }
222  else
223  {
224  for(bin_size_type i=0;i<N_bins;++i)
225  {
226  bins.push_back(bin(x_min+i*bin_size,0,0));
227  }
228  }
229 
230  /* Filling bins: */
231 
232  if(weight==NULL)
233  {
234  for(data_size_type i=0;i<N_points;++i)
235  {
236  bin b(data[i].value());
237  typename std::vector<bin>::iterator it=std::lower_bound(bins.begin(),bins.end(),b,bin_pos_comp);
238  if(it != bins.begin() and it != bins.end())
239  {
240  ++it->value();
241  }
242  }
243  }
244  else
245  {
246  for(data_size_type i=0;i<N_points;++i)
247  {
248  bin b(data[i].value(),0,0);
249  typename std::vector<bin>::iterator it=std::lower_bound(bins.begin(),bins.end(),b,bin_pos_comp);
250  if(it != bins.begin() and it != bins.end())
251  {
252  ++it->value();
253  it->weight()+=data[i].weight();
254  }
255  }
256  value_type norm=N_points/weight_sum;
257  for(bin_size_type i=0;i<N_bins;++i)
258  {
259  bins[i].weight()*=norm;
260  }
261  }
262  max_freq=std::max_element(bins.begin(),bins.end(),bin_value_comp)->value();
263  }
