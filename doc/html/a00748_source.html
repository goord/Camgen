53  static generator_type* create_generator(tree_iterator it,is_type is_type_,fs_type fs_type_)
54  {
55  bool backward_shat=(fs_type_==phase_space_generators::recursive_backward_shat);
56  is_generator_type* isgen=is_factory_type::create_instance(is_type_,backward_shat);
57  if(isgen==NULL)
58  {
59  return NULL;
60  }
61  generator_type* result=fs_factory_type::create_instance(isgen,fs_type_);
62  if(result!=NULL)
63  {
64  if(!result->set_amplitude(it))
65  {
66  return NULL;
67  }
68  for(int i=1;i<=(int)N_in;++i)
69  {
70  result->set_beam_energy(-i,beam_energy(i));
71  }
72  result->refresh_Ecm();
73  for(int i=1;i<=(int)N_out;++i)
74  {
75  for(int j=i+1;j<=(int)N_out;++j)
76  {
77  result->set_m_min(i,j,basic_cuts::m_min(i,j));
78  }
79  }
80  result->refresh_m_min();
81  }
82  return result;
83  }
