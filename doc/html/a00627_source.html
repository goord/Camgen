25  template<std::size_t N_in,std::size_t N_out>bool decompose_process(const std::string& proc,vector<std::string,N_in+N_out>& vec)
26  {
27  typename std::string::size_type counter=0;
28  bool in=true;
29  typename std::string::const_iterator it=proc.begin();
30  while(it != proc.end())
31  {
32  if(counter==(N_in+N_out))
33  {
34  log(log_level::warning)<<CAMGEN_STREAMLOC<<"detected more than "<<N_out<<" final state particles for "<<proc<<endlog;
35  return true;
36  }
37  if(*it==' ')
38  {
39  ++it;
40  continue;
41  }
42  if(*it==',')
43  {
44  if(counter==(N_in-1) and in)
45  {
46  log(log_level::warning)<<CAMGEN_STREAMLOC<<"detected more than "<<N_in<<" initial state particles for "<<proc<<endlog;
47  it=std::find(it,proc.end(),'>');
48  }
49  else
50  {
51  ++counter;
52  ++it;
53  }
54  continue;
55  }
56  if(*it=='>')
57  {
58  if(counter!=(N_in-1))
59  {
60  log(log_level::warning)<<CAMGEN_STREAMLOC<<"detected less than "<<N_in<<" initial state particles for "<<proc<<endlog;
61  return false;
62  }
63  ++counter;
64  ++it;
65  in=false;
66  continue;
67  }
68  vec[counter].push_back(*it);
69  ++it;
70  }
71  if(counter<(N_in+N_out-1))
72  {
73  log(log_level::warning)<<"detected less than "<<N_out<<" final state particles for "<<proc<<endlog;
74  return false;
75  }
76  return true;
77  }
78 }
