101  static void evaluate(const momentum_type* p,const r_value_type* m,const r_value_type* w)
102  {
103  CAMGEN_ERROR_IF((p==NULL),"NULL momentum pointer encountered");
104 
105  momentum=p;
106  mass=m;
107  width=w;
108  s=spacetime_type::dot(*p,*p);
109 
110  if(m==NULL)
111  {
112  CAMGEN_MESSAGE_IF((s==0),"On-shell massless propagator encountered");
113 
114  denominator=value_type(0,1)/s;
115  fermion_mass=value_type(0,0);
116  gauge_mass2=s;
117  return;
118  }
119  if(w==NULL or !switched_on)
120  {
121  CAMGEN_MESSAGE_IF((s==(*m)*(*m)),"On-shell zero-width propagator encountered");
122 
123  denominator=value_type(0,1)/(s-(*m)*(*m));
124  fermion_mass=value_type(*m,0);
125  gauge_mass2=value_type((*m)*(*m),0);
126  return;
127  }
128  if(s<(r_value_type)0 and !spacelike_widths)
129  {
130  CAMGEN_MESSAGE_IF((s==(*m)*(*m)),"On-shell spacelike propagator encountered");
131 
132  denominator=value_type(0,1)/(s-(*m)*(*m));
133  fermion_mass=value_type(*m,0);
134  gauge_mass2=value_type((*m)*(*m),0);
135  return;
136  }
137  if(running_widths)
138  {
139  denominator=value_type(0,1)/value_type(s-(*m)*(*m),(*w)*s/(*m));
140  }
141  else
142  {
143  denominator=value_type(0,1)/value_type(s-(*m)*(*m),(*w)*(*m));
144  }
145  if(complex_masses)
146  {
147  gauge_mass2=value_type((*m)*(*m),-(*w)*(*m));
148  fermion_mass=std::sqrt(gauge_mass2);
149  }
150  else
151  {
152  gauge_mass2=value_type((*m)*(*m),0);
153  fermion_mass=value_type(*m,0);
154  }
155  }
