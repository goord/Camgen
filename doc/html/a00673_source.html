104  static generator_type* create_instance(initial_state_type is_type,bool backward_shat)
105  {
106  switch(is_type)
107  {
108  case initial_states::partonic:
109  return new partonic_is<model_t,2,Minkowski_type>;
110  case initial_states::eplus_eminus:
111  return new partonic_is<model_t,2,Minkowski_type>;
112  case initial_states::proton_proton:
113  if(backward_shat)
114  {
115  return new hadronic_is_y<model_t,rng_t,Minkowski_type>(false,false);
116  }
117  else
118  {
119  return new hadronic_is_sy<model_t,rng_t,Minkowski_type>(false,false);
120  }
121  case initial_states::proton_antiproton:
122  if(backward_shat)
123  {
124  return new hadronic_is_y<model_t,rng_t,Minkowski_type>(false,true);
125  }
126  else
127  {
128  return new hadronic_is_sy<model_t,rng_t,Minkowski_type>(false,true);
129  }
130  case initial_states::antiproton_proton:
131  if(backward_shat)
132  {
133  return new hadronic_is_y<model_t,rng_t,Minkowski_type>(true,false);
134  }
135  else
136  {
137  return new hadronic_is_sy<model_t,rng_t,Minkowski_type>(true,false);
138  }
139  case initial_states::antiproton_antiproton:
140  if(backward_shat)
141  {
142  return new hadronic_is_y<model_t,rng_t,Minkowski_type>(true,true);
143  }
144  else
145  {
146  return new hadronic_is_sy<model_t,rng_t,Minkowski_type>(true,true);
147  }
148  case initial_states::proton_proton_xx:
149  return new hadronic_is_xx<model_t,rng_t,Minkowski_type>(false,false);
150  case initial_states::proton_antiproton_xx:
151  return new hadronic_is_xx<model_t,rng_t,Minkowski_type>(false,true);
152  case initial_states::antiproton_proton_xx:
153  return new hadronic_is_xx<model_t,rng_t,Minkowski_type>(true,false);
154  case initial_states::antiproton_antiproton_xx:
155  return new hadronic_is_xx<model_t,rng_t,Minkowski_type>(true,true);
156  default:
157  return NULL;
158  }
159  }
