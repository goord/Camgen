100  static void initialise(event_generator_type* evtgen,generator_configuration<model_t>& conf,bool verbose=false)
101  {
102  if(evtgen==NULL)
103  {
104  return;
105  }
106 
107  typename event_generator_type::process_container& procs=evtgen->procs;
108  typename event_generator_type::process_iterator& sub_proc=evtgen->sub_proc;
109 
110  for(sub_proc=evtgen.procs.begin();sub_proc!=evtgen.procs.end();++sub_proc)
111  {
112  if(verbose)
113  {
114  std::stringstream ss;
115  (*sub_proc)->print_process(ss);
116  std::cout<<std::endl<<"init subprocess "<<ss.str()<<std::endl;
117  }
118  conf.configure();
119  procgen_fac_type::initialise(&(*sub_proc),verbose);
120  sub_proc.loop_process(subprocess_events(),verbose);
121  }
122  evtgen->refresh_cross_section();
123  evtgen->adapt_processes();
124  }

130  void configure(event_generator_type* evtgen,generator_configuration<model_t>* conf)
131  {
132  typedef typename model_t::value_t value_type;
133 
134  amplitude_type& amplitude=evtgen.algorithm;
135  typename event_generator_type::process_container& procs=evtgen->procs;
136 
137 
138  procs.clear();
139  procs.reserve(amplitude.n_trees());
140  if(amplitude.reset_process())
141  {
142  std::size_t id=0;
143  do
144  {
145  if(!amplitude.get_tree_iterator()->is_empty())
146  {
147  procs.push_back(process_generator_factory.create_generator(amplitude.get_tree_iterator(),conf,id++));
148  }
149  }
150  while(amplitude.next_process());
151  if(procs.size()!=0)
152  {
153  value_type alpha=(value_type)1/(value_type)procs.size();
154  for(std::size_t i=0;i<procs.size();++i)
155  {
156  procs[i]->alpha=alpha;
157  }
158  evtgen->sub_proc=procs.begin();
159  }
160  else
161  {
162  evtgen->sub_proc=procs.end();
163  return;
164  }
165  }
166  else
167  {
168  evtgen->sub_proc=procs.end();
169  return;
170  }
171  evtgen->sub_proc=procs.begin();
172  if(conf!=NULL)
173  {
174  conf->lock_generator(evtgen);
175  conf->configure();
176  }
177  }
