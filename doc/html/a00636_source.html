32  template<class value_type>value_type offshell_VV_width(std::complex<value_type>& ymc,const value_type& mh,const value_type& m1,const value_type& w1,const value_type& m2,const value_type& w2,const value_type& eps=0.01)
33  {
34  value_type mw1=m1*w1;
35  value_type mw12=mw1*mw1;
36  value_type m12=m1*m1;
37  value_type mw2=m2*w2;
38  value_type mw22=mw2*mw2;
39  value_type m22=m2*m2;
40  value_type mh2=mh*mh;
41  BW_s_generator<value_type,std::random>gen1(&m1,&w1),gen2(&m2,&w2);
42  gen1.set_m_range(0,mh);
43  gen2.set_m_range(0,mh);
44  const value_type& s1=gen1.s();
45  const value_type& s2=gen2.s();
46  value_type wght(0),wsum(0),w2sum(0),err(1);
47  std::size_t N=0,n=0;
48  do
49  {
50  bool q,q1,q2;
51  n=0;
52  do
53  {
54  q1=gen1.generate();
55  q2=gen2.generate();
56  q=(q1 and q2 and (std::sqrt(s1)+std::sqrt(s2)<=mh));
57  ++n;
58  }
59  while(!q and (n<max_s_pairs()));
60  if(!q)
61  {
62  continue;
63  }
64  else
65  {
66  ++N;
67  value_type num=((value_type)2+std::pow(mh2-s1-s2,(int)2)/(4*m12*m22))*std::sqrt(Kallen(mh2,s1,s2));
68  value_type denom=(std::pow(s1-m12,(int)2)+mw12)*(std::pow(s2-m22,(int)2)+mw22);
69  wght=integrate(&gen1,&gen2,mh)*gen1.weight()*gen2.weight()*num/denom;
70  wsum+=wght;
71  w2sum+=(wght*wght);
72  }
73  err=(wsum==0 or N<2)?1:(std::sqrt(w2sum-wsum*wsum/N)/wsum);
74  }
75  while(err>eps);
76  value_type pi=std::acos(-(value_type)1);
77  return mw1*mw2*std::norm(ymc)*(wsum/N)/std::pow(2*pi*mh,(int)3);
78  }
