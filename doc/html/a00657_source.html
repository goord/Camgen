110  /* Generation method implementation. */
111 
112  bool generate()
113  {
114  if(!xgen->generate())
115  {
116  this->weight()=(value_type)0;
117  return false;
118  }
119 
120  value_type P1=xvec[0]*this->beam_energy(0);
121  value_type P2=xvec[1]*this->beam_energy(1);
122 
123  flip=rn_stream::throw_coin();
124  size_type pa=flip?0:1;
125  size_type pb=flip?1:0;
126 
127  value_type Ehat1=std::sqrt(P1*P1+this->m2(pa));
128  value_type Ehat2=std::sqrt(P2*P2+this->m2(pb));
129  if(!this->set_s_hat(this->m2(0)+this->m2(1)+(value_type)2*(Ehat1*Ehat2+P1*P2)))
130  {
131  this->weight()=(value_type)0;
132  this->integrand()=(value_type)0;
133  return false;
134  }
135 
136  this->p(pa).assign(0);
137  this->p(pa)[k0]=Ehat1;
138  this->p(pa)[kL]=P1;
139 
140  this->p(pb).assign(0);
141  this->p(pb)[k0]=Ehat2;
142  this->p(pb)[kL]=-P2;
143 
144  this->weight()=xgen->weight();
145  return true;
146  }
147 
148  /* Weight evaluation implementation: */
149 
150  bool evaluate_weight()
151  {
152  if(!xgen->evaluate_weight())
153  {
154  this->weight()=(value_type)0;
155  return false;
156  }
157  const momentum_type& p1=this->p(0);
158  const momentum_type& p2=this->p(1);
159  if(p1[0]<this->m(0) or p2[0]<this->m(1))
160  {
161  this->weight()=(value_type)0;
162  this->integrand()=(value_type)0;
163  return false;
164  }
165  value_type P1=std::abs(p1[kL]);
166  value_type P2=std::abs(p2[kL]);
167  if(!this->set_s_hat(this->m2(0)+this->m2(1)+(value_type)2*(p1[0]*p2[0]+P1*P2)))
168  {
169  this->weight()=(value_type)0;
170  this->integrand()=(value_type)0;
171  return false;
172  }
173  xvec[0]=P1/this->beam_energy(0);
174  xvec[1]=P2/this->beam_energy(1);
175  if(!xgen->evaluate_weight())
176  {
177  this->weight()=(value_type)0;
178  return false;
179  }
180  this->weight()=xgen->weight();
181  return true;
182  }
