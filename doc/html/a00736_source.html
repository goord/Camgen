42  template<class value_t,class rng_t>value_t integrate(const BW_sampler<value_t,rng_t>& bw1,const BW_sampler<value_t,rng_t>& bw2,const value_t& s)
43  {
44  value_t w=std::sqrt(s)-bw1.sqrtsmin-bw2.sqrtsmin;
45  if(w<(value_t)0)
46  {
47  return (value_t)0;
48  }
49  std::complex<value_t>result;
50  result = BWhelper(bw1.nu/(bw1.nu+bw2.nu-w),bw2.nu/(bw1.nu+bw2.nu-w))+BWhelper(bw1.nu/(bw1.nu-bw2.nu+w),bw2.nu/(-bw1.nu+bw2.nu-w))+BWhelper(bw1.nu/(bw1.nu-bw2.nu-w),bw2.nu/(-bw1.nu+bw2.nu+w))+BWhelper(bw1.nu/(bw1.nu+bw2.nu+w),bw2.nu/(bw1.nu+bw2.nu+w)) ;
51  result+=(BWhelper(bw1.nubar/(bw1.nubar+bw2.nubar-w),bw2.nubar/(bw1.nubar+bw2.nubar-w))+BWhelper(bw1.nubar/(bw1.nubar-bw2.nubar+w),bw2.nubar/(-bw1.nubar+bw2.nubar-w))+BWhelper(bw1.nubar/(bw1.nubar-bw2.nubar-w),bw2.nubar/(-bw1.nubar+bw2.nubar+w))+BWhelper(bw1.nubar/(bw1.nubar+bw2.nubar+w),bw2.nubar/(bw1.nubar+bw2.nubar+w)));
52  result-=(BWhelper(bw1.nu/(bw1.nu+bw2.nubar-w),bw2.nubar/(bw1.nu+bw2.nubar-w))+BWhelper(bw1.nu/(bw1.nu-bw2.nubar+w),bw2.nubar/(-bw1.nu+bw2.nubar-w))+BWhelper(bw1.nu/(bw1.nu-bw2.nubar-w),bw2.nubar/(-bw1.nu+bw2.nubar+w))+BWhelper(bw1.nu/(bw1.nu+bw2.nubar+w),bw2.nubar/(bw1.nu+bw2.nubar+w)));
53  result-=(BWhelper(bw1.nubar/(bw1.nubar+bw2.nu-w),bw2.nu/(bw1.nubar+bw2.nu-w))+BWhelper(bw1.nubar/(bw1.nubar-bw2.nu+w),bw2.nu/(-bw1.nubar+bw2.nu-w))+BWhelper(bw1.nubar/(bw1.nubar-bw2.nu-w),bw2.nu/(-bw1.nubar+bw2.nu+w))+BWhelper(bw1.nubar/(bw1.nubar+bw2.nu+w),bw2.nu/(bw1.nubar+bw2.nu+w)));
54  return -(value_t)0.25*result.real()/(bw1.mw*bw2.mw);
55  }

60  template<class value_t,class rng_t>value_t integrate(const pl_sampler<value_t,rng_t>& pl1,const pl_sampler<value_t,rng_t>& pl2,const value_t& s)
61  {
62  value_t zmin1=std::sqrt(pl1.smin()/s);
63  value_t zmin2=std::sqrt(pl2.smin()/s);
64  if(zmin1+zmin2>(value_t)1)
65  {
66  return (value_t)0;
67  }
68  value_t zplus2=(value_t)1-zmin2;
69  value_t a=(value_t)2*((value_t)1-*(pl1.nu));
70  value_t b=(value_t)2*((value_t)1-*(pl2.nu));
71  value_t integral=(value_t)0;
72  value_t up =(value_t)0.5*(zplus2+zmin1);
73  value_t low=(value_t)0.5*(zplus2-zmin1);
74  value_t x;
75  for(unsigned n=0;n<GAUSS_INT_ORDER;++n)
76  {
77  x=Legendre<value_t,GAUSS_INT_ORDER>::root(n)*low+up;
78  integral+=Legendre<value_t,GAUSS_INT_ORDER>::weight(n)*std::pow(x,a-(value_t)1)*std::pow(((value_t)1-x),b);
79  }
80  integral*=low;
81  value_t prefactor=(value_t)4*std::pow(s,(value_t)0.5*(a+b))/b;
82  return prefactor*(integral-std::pow(zmin2,b)*(std::pow(zplus2,a)-std::pow(zmin1,a))/a);
83  }

96  template<class value_t,class rng_t>value_t integrate(const BW_sampler<value_t,rng_t>& bw,const pl_sampler<value_t,rng_t>& pl,const value_t& s)
97  {
98  value_t sqrts=std::sqrt(s);
99  value_t sqrtsminbw=std::sqrt(bw.smin());
100  value_t sqrtsminpl=std::sqrt(pl.smin());
101  if(sqrts<=sqrtsminbw+sqrtsminpl)
102  {
103  return (value_t)0;
104  }
105  value_t nuplus=(value_t)1-*(pl.nu);
106  value_t low=std::pow(sqrts-sqrtsminpl,(value_t)2*nuplus);
107  value_t mid=std::pow(sqrts-*(bw.m),(value_t)2*nuplus);
108  value_t integral=(value_t)0;
109  value_t umin2=((value_t)1-*(pl.nu))*pl.r_min;
110  value_t x,y,z,I1,I2;
111 
112  for(unsigned n=0;n<GAUSS_INT_ORDER;++n)
113  {
114  x=(value_t)0.5*(Legendre<value_t,GAUSS_INT_ORDER>::root(n)+(value_t)1);
115  y=std::pow(sqrts-std::pow(low+(mid-low)*x,(value_t)0.5*pl.inv_exp),2);
116  I1=std::atan((y-bw.mm)/bw.mw)-bw.r_min;
117  z=std::pow(sqrts-std::pow(umin2-(umin2-mid)*x,(value_t)0.5*pl.inv_exp),2);
118  I2=std::atan((z-bw.mm)/bw.mw)-bw.r_min;
119  integral+=Legendre<value_t,GAUSS_INT_ORDER>::weight(n)*((mid-low)*I1+(umin2-mid)*I2);
120  }
121  return -(value_t)0.5*integral/(bw.mw*nuplus);
122  }

185  template<class value_t,class rng_t>value_t integrate(const u_sampler<value_t,rng_t>& u,const pl_sampler<value_t,rng_t>& pl,const value_t& s)
186  {
187  value_t sqrts=std::sqrt(s);
188  value_t a=sqrts-std::sqrt(u.smin_val);
189  value_t b=std::sqrt(pl.smin());
190  value_t fac=(value_t)2-(value_t)2*(*pl.nu);
191  value_t A=std::pow(a,fac);
192  value_t B=std::pow(b,fac);
193  value_t result=(value_t)2*(s-u.smin_val)*(A-B)/fac;
194  ++fac;
195  A*=a;
196  B*=b;
197  result-=((value_t)4*sqrts*(A-B)/fac);
198  ++fac;
199  A*=a;
200  B*=b;
201  result+=((value_t)2*(A-B)/fac);
202  return result;
203  }

213  template<class value_t,class rng_t>value_t integrate(const u_sampler<value_t,rng_t>& u,const BW_sampler<value_t,rng_t>& bw,const value_t& s)
214  {
215  value_t sqrts=std::sqrt(s);
216  value_t sqrts1=sqrts-std::sqrt(u.smin_val);
217  value_t sqrts2=std::sqrt(bw.smin());
218  value_t seff=sqrts1*sqrts1;
219  std::complex<value_t>lterm=bw.nu*std::log(((bw.nu+sqrts1)*(bw.nu-sqrts2))/((bw.nu-sqrts1)*(bw.nu+sqrts2)));
220  std::complex<value_t>lbarterm=bw.nubar*std::log(((bw.nubar+sqrts1)*(bw.nubar-sqrts2))/((bw.nubar-sqrts1)*(bw.nubar+sqrts2)));
221 
222  return ((s-u.smin_val+bw.mm)/bw.mw)*(std::atan((seff-bw.mm)/bw.mw)-bw.r_min)+(value_t)0.5*std::log(((seff-bw.mm)*(seff+bw.mm)+bw.mw*bw.mw)/(std::pow(bw.smin()-bw.mm,2)+bw.mw*bw.mw))+sqrts*(lterm.real()+lbarterm.real())/bw.mw;
223  }
