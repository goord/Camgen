44  static s_generator_type* create_s_generator(const particle_channel_type* channel,const value_type* Ecmhat=NULL)
45  {

47  const value_type* mass=(phi==NULL)?NULL:(phi->get_mass_address());
48  const value_type* width=(phi==NULL or channel->spacelike())?NULL:(phi->get_width_address());
49  const value_type* exponent=&(channel->s_sampling_exponent);

51  bool adaptive=(channel->timelike())?adaptive_s_sampling():adaptive_t_sampling();

58  }

238  static branching_type* create_t_branching(particle_channel_type* in1,particle_channel_type* in2,particle_channel_type* out1,particle_channel_type* out2,particle_channel_type* out3,particle_channel_type* out4)
239  {
240  bit_string_type sa=in1->bitstring();
241  bit_string_type sb=in2->bitstring();
242  bit_string_type s1=out1->bitstring();
243  bit_string_type st=out2->bitstring();
244  bit_string_type s2=out3->bitstring();
245  bit_string_type stot=out4->bitstring();
246 
247  if(!sa[0])
248  {
249  log(log_level::warning)<<CAMGEN_STREAMLOC<<"incoming channel "<<sa<<" cannot be part of a t-channel--returning NULL"<<endlog;
250  return NULL;
251  }
252  if(!(sb[0] and sb.count()==1))
253  {
254  log(log_level::warning)<<CAMGEN_STREAMLOC<<"second incoming momentum "<<sb<<" is not the second beam momentum--returning NULL"<<endlog;
255  return NULL;
256  }
257  if(s1[0])
258  {
259  log(log_level::warning)<<CAMGEN_STREAMLOC<<"first outgoing channel "<<s1<<" is not timelike--returning NULL"<<endlog;
260  return NULL;
261  }
262  if((st & s1).any())
263  {
264  log(log_level::warning)<<CAMGEN_STREAMLOC<<"invalid combination of t-channel "<<st<<" and first s-channel "<<s1<<" ignored--returning NULL"<<endlog;
265  return NULL;
266  }
267  if((st | s1)!=sa)
268  {
269  log(log_level::warning)<<CAMGEN_STREAMLOC<<"invalid combination of t-channel "<<st<<" and first s-channel "<<s1<<" and incoming channel "<<sa<<" ignored--returning NULL"<<endlog;
270  return NULL;
271  }
272  if((s1 | s2)!=stot)
273  {
274  log(log_level::warning)<<CAMGEN_STREAMLOC<<"total outgoing channel "<<stot<<" is not the combination of the outgoing channels "<<s1<<" and "<<s2<<"--returning NULL"<<endlog;
275  return NULL;
276  }
277  bit_string_type bt=st;
278  bt.reset(0);
279  if(bt!=s2)
280  {
281  log(log_level::warning)<<CAMGEN_STREAMLOC<<"invalid combination of t-channel "<<st<<" and second s-channel "<<s2<<" ignored--returning NULL"<<endlog;
282  return NULL;
283  }
284  return new t_branching<model_t,N_out,rng_t,spacetime_type>(in1,in2,out1,out2,out3,out4,false);
285  }
