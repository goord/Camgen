167  value_type max_weight(const value_type& epsilon) const
168  {
169  if(!(epsilon>(value_type)0))
170  {
171  return maxw;
172  }
173  if(!(epsilon<(value_type)1))
174  {
175  return 0;
176  }
177  typename std::vector<value_type>::const_reverse_iterator it=freqs.rbegin();
178  size_type i=freqs.size()-1;
179  value_type bw(bin_width()),bound(epsilon*wsum/bw),Splus((*it)*i);
180  while(Splus<bound and it!=freqs.rend())
181  {
182  ++it;
183  --i;
184  Splus+=((*it)*i);
185  }
186  return (i-1+(Splus-bound)/(i*(*it)))*bw;
187  }

273  value_type integrate(value_type range,iterator end)
274  {
275  if(pos==end)
276  {
277  return 0;
278  }
279  if(range+offset<1)
280  {
281  offset+=range;
282  return range*(*pos);
283  }
284  value_type r=range+offset-1;
285  value_type result=(1-offset)*(*pos);
286  size_type n=static_cast<size_type>(std::floor(r));
287  for(size_type i=0;i<n;++i)
288  {
289  if((++pos)==end)
290  {
291  break;
292  }
293  result+=(*pos);
294  }
295  if(pos!=end)
296  {
297  offset=r-n;
298  ++pos;
299  result+=(offset*(*pos));
300  }
301  else
302  {

304  }

312  /* Rebins such that the new maximal weight w is contained in the
313  * last bin and the number of bins is not changed. */
314 
315  void rebin(const value_type& w)
316  {
317  if(w<maxw)
318  {
319  return;
320  }
321  value_type oldwidth=bin_width();
322  value_type newwidth=w/(freqs.size()-1);
323  std::vector<value_type>newfreqs(freqs.size());
324  bin_iterator it(freqs.begin());
325  for(size_type i=0;i<newfreqs.size()-1;++i)
326  {
327  newfreqs[i]=it.integrate(newwidth/oldwidth,freqs.end());
328  }
329  value_type lastfill=it.integrate(newwidth/oldwidth,freqs.end());
330  if(lastfill>(value_type)0)
331  {
332  log(log_level::warning)<<CAMGEN_STREAMLOC<<"the last bin is incorrectly filled with "<<lastfill<<endlog;
333  }
334  freqs=newfreqs;
335  }
