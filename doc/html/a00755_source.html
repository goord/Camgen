15 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
16  * Declaration and definition of the R_xi gauge vector and scalar propagators. *
17  * In the current version of Camgen, all R_xi propagators in a process use the *
18  * same value of xi, defined in the class R_gauge<model_type>, and by default *
19  * set to zero. The vector propagator invoke the formula *
20  * *
21  * P(p)^{mu,nu} = -i(g^{mu,nu} - (1-xi)*p^{mu}p^{nu}/(p^2-xi*M^2)/(p^2-M^2) *
22  * *
23  * where M^2=m^2-i*m*Gamma, Gamma being the width. For the scalar propagator, *
24  * which is used by the would-be Goldstones of a broken symmetry, the xi appears *
25  * in front of the mass squared in the denominator, making them decouple as xi *
26  * approaches infinity. *
27  * *
28  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */

107  /* Evaluate function: propagates all vectors between 'first' and 'last': */
108 
109  static void evaluate(iterator first,iterator last)
110  {
111  const momentum_type* p=width_scheme<model_t>::momentum;
112  CAMGEN_ERROR_IF((p==NULL),"attempt to dereference a NULL momentum pointer");
113  CAMGEN_ERROR_IF((first.range()<model_t::dimension),"tensor iterator out of range");
114  CAMGEN_ERROR_IF(((last-first)<0),"bounding iterator smaller than running iterator");
115  CAMGEN_ERROR_IF(((last-first)%model_t::dimension != 0),"first and last iterator differ by non-integer multiple of vector size");
116  value_type p2(width_scheme<model_t>::s,0);
117  value_type r=value_type(1-R_gauge<model_t>::xi,0)/(p2-R_gauge<model_t>::xi*width_scheme<model_t>::gauge_mass2);
118  value_type a;
119  do
120  {
121  a=spacetime_type::dot(*p,first)*r;
122  for(size_type mu=0;mu<model_t::dimension;++mu)
123  {
124  (*first)-=a*(*p)[mu];
125  (*first)*=(-width_scheme<model_t>::denominator);
126  ++first;
127  }
128  }
129  while(first!=last);
130  }
