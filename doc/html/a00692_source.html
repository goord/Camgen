168  template<class rng_t>void generate_unweighted()
169  {
170  if(!this->validate_integrand())
171  {
172  return;
173  }
174  value_type rho,x;
175  do
176  {
177  this->generate();
178  x=weighted_integrand();

246  virtual MC_integral<value_t> cross_section() const
247  {
248  if(up_to_date)
249  {
250  return integral;
251  }
252  value_type N=(value_type)n_calls;
253  value_type Ninv=(value_type)1/N;
254  value_type result=(n_calls!=0)?Ninv*wsum:(value_type)0;
255  MC_integral<value_type>sigma(result);
256  value_type N2=N*(N-(value_type)1);
257  value_type E2=(w2sum-Ninv*wsum*wsum)/N2;
258  sigma.error=(n_calls>1)?std::sqrt(E2):(std::numeric_limits<value_type>::infinity());
259  value_type N4=N2*(N-(value_type)2)*(N-(value_type)3);
260  value_type N24=N2/N4;
261  sigma.error_error=(n_calls>3)?std::pow(N24*Ninv*Ninv*(w4sum-(value_type)4*Ninv*w3sum*wsum+(value_type)3*Ninv*w2sum*w2sum)+((value_type)1-N2*N24)*E2*E2,0.25):(std::numeric_limits<value_type>::infinity());
262  integral=sigma;
263  up_to_date=true;
264  return sigma;
265  }
