14 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
15  * Minkowski spacetime definition. For models in dimensions > 0, a spacetime type *
16  * definition is required for the inner products of momenta and Lorents vectors. A *
17  * spacetime class wraps an implementation metafunction, taking the value type and *
18  * dimension as parameters and defining the inner products between real- or *
19  * complex-values Lorentz vectors. Moreover, the spacetime implementation class *
20  * contains functions for constructing the base vectors k_0 and k_1 for massles basic*
21  * spinors and functions that construct spin vectors for massive spinors. The *
22  * Minkowski metric in Camgen has signature (+ - - -). *
23  * *
24  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */

163  {

180  /* Function defining the construction of the polarised spin
181  * vector. It is defined as
182  *
183  * s=e - (e.p/(m*(E + m))*(p + p0)
184  *
185  * where e is the spacelike unit vector in the polarisation
186  * direction and p0 the rest-frame four-momentum. Note that
187  * in the rest-frame s=e.
188  * */

190  template<int beam_dir>static vector<value_t,dim>& make_polarised_spinvec(vector<value_t,dim>& s,const vector<value_t,dim>& p,const value_t& m)
191  {
192  CAMGEN_MESSAGE_IF((m<=(value_t)0),"zero or negative mass encountered");
193  CAMGEN_MESSAGE_IF((p[0]==m),"center-of-mass momentum encountered");
194 
195  if(beam_dir>0)
196  {
197  s[0]=p[beam_dir]/m;
198  value_t a=s[0]/(p[0]+m);
199  for(int i=1;i<dim;++i)
200  {
201  s[i]=a*p[i];
202  }
203  s[beam_dir]+=((value_t)1);
204  }
205  if(beam_dir<0)
206  {
207  s[0]=-p[-beam_dir]/m;
208  value_t a=s[0]/(p[0]+m);
209  for(int i=1;i<dim;++i)
210  {
211  s[i]=a*p[i];
212  }
213  s[-beam_dir]-=((value_t)1);
214  }
215  return s;
216  }

221  static void split_momentum(vector<value_t,dim>& p1,vector<value_t,dim>& p2,const vector<value_t,dim>& p,const value_t& m)
222  {
223  CAMGEN_MESSAGE_IF((m<=(value_t)0),"zero or negative mass encountered");

225

228  value_t a=0.5*msq/(pvec*(p[0]-pvec));

235  p2-=p1;
236  }
