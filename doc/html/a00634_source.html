268  bool generate()
269  {
270  if(procs.size()==0)
271  {
272  this->weight()=(value_type)0;
273  return false;
274  }
275  value_type rho=throw_number((value_type)0,(value_type)1);
276  sub_proc=procs.begin();
277  value_type r(0);
278  while(r<=rho and sub_proc!=procs.end())
279  {
280  r+=(*sub_proc)->alpha;
281  ++sub_proc;
282  }
283  --sub_proc;
284  if(sub_proc==procs.end())
285  {
286  log(log_level::warning)<<CAMGEN_STREAMLOC<<"subprocess iterator overflow detected--no generation performed"<<endlog;
287  return false;
288  }
289  bool q=(*sub_proc)->generate();
290  (*sub_proc)->refresh_cross_section();
291  this->integrand()=(*sub_proc)->integrand();
292  this->weight()=(*sub_proc)->weight()/(*sub_proc)->alpha;
293  if(update_counter!=0 and auto_proc_adapt!=0 and update_counter%auto_proc_adapt==0)
294  {
295  adapt_processes();
296  }
297  return q;
298  }

460  void adapt_processes()
461  {
462  value_type norm(0);
463  MC_integral<value_type>proc_xsec;
464  value_type sigma=cross_section_sum().value;
465  value_type q=process_threshold*sigma/procs.size();
466  for(process_iterator it=procs.begin();it!=procs.end();++it)
467  {
468  proc_xsec=(*it)->cross_section();
469  if((proc_xsec.value+proc_xsec.error)>q)
470  {
471  (*it)->alpha=std::pow((*it)->cross_section().error,process_adaptivity);
472  norm+=(*it)->alpha;
473  }
474  else
475  {
476  std::cout<<"deleting process with xsec "<<proc_xsec<<" smaller than "<<q<<std::endl;
477  (*it)->alpha=(value_type)0;
478  }
479  }
480  for(process_iterator it=procs.begin();it!=procs.end();++it)
481  {
482  (*it)->alpha/=norm;
483  }
484  std::sort(procs.begin(),procs.end(),alpha_more);
485  process_iterator it=procs.begin();
486  while(it!=procs.end() and (*it)->alpha!=(value_type)0)
487  {
488  ++it;
489  }
490  if(it==procs.end() or it==procs.begin())
491  {
492  return;
493  }
494  std::size_t newsize=it-procs.begin();
495  process_iterator it2;
496  while(it!=procs.end())
497  {
498  algorithm.set_process((*it)->amplitude);
499  algorithm.remove_process();
500  delete (*it);
501  ++it;
502  }
503  procs.resize(newsize);
504  sub_proc=procs.begin();
505  }

534  cross_section_type cross_section_sum() const
535  {
536  if(!this->up_to_date)
537  {
538  summed_xsec.value=(value_type)0;
539  summed_xsec.error=(value_type)0;
540  summed_xsec.error_error=(value_type)0;
541 
542  for(const_process_iterator it=procs.begin();it!=procs.end();++it)
543  {
544  MC_integral<value_type>xsect((*it)->cross_section());
545  summed_xsec+=xsect.value;
546  summed_xsec.error+=(xsect.error*xsect.error);
547  summed_xsec.error_error+=(xsect.error_error*xsect.error_error);
548  }
549  summed_xsec.error=std::sqrt(summed_xsec.error);
550  summed_xsec.error_error=std::sqrt(summed_xsec.error_error);
551  this->up_to_date=true;
552  }
553  return summed_xsec;
554  }
