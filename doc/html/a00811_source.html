38  /* Initialisation phase, creating the vector of 'jumps' of the iterators
39  * to nonzero entries: */
40 
41  static void initialise()
42  {
43  if(!initialised)
44  {
45  size_type Isize=Feynrule_t::sizes[I];
46  size_type Jsize=Feynrule_t::sizes[J];
47  size_type Ksize=Feynrule_t::sizes[K];
48  size_type Lsize=Feynrule_t::sizes[L];
49  rep_type::initialise();
50 
51  /* First, list the nonzero components in the 'jumps' vectors: */
52 
53  value_type temp;
54  for(size_type a=0;a<rep_t::group_type::rank;++a)
55  {
56  for(size_type b=0;b<rep_t::group_type::rank;++b)
57  {
58  for(size_type i=0;i<rep_t::dimension;++i)
59  {
60  for(size_type j=0;j<rep_t::dimension;++j)
61  {
62  temp=value_type(0,0);
63  for(size_type k=0;k<rep_t::dimension;++k)
64  {
65  temp+=((rep_type::generator(a,i,k))*(rep_type::generator(b,k,j)));
66  }
67  if(temp != value_type(0,0))
68  {
69  values.push_back(temp);
70  jumps[0].push_back(a*Isize);
71  jumps[1].push_back(b*Jsize);
72  jumps[2].push_back(i*Ksize);
73  jumps[3].push_back(j*Lsize);
74  }
75  }
76  }
77  }
78  }
79 
80  /* Secondly, we subtract each jump entry by the previous one to
81  * obtain the offsets: */
82 
83  for(size_type i=0;i<4;++i)
84  {
85  backsets[i]=jumps[i].back();
86  for(int j=size()-1;j!=0;--j)
87  {
88  jumps[i][j]-=jumps[i][j-1];
89  }
90  }
91  initialised=true;
92  }
93  }
94 
95  /* Number of nonzero components: */
96 
97  static size_type size()
98  {
99  return values.size();
100  }

135  }
