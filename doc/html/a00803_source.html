95  bool generate_s()
96  {
97  sweight=(value_type)1;
98  if(this->backward_s_sampling)
99  {
100  if(stot_channel->get_status()==ps_channel_type::reset)
101  {
102  stot_channel->shortprint(std::cerr);
103  log(log_level::warning)<<CAMGEN_STREAMLOC<<"sampling invariant mass without total invariant mass generated..."<<endlog;
104  }
105  if(!s1_channel->on_shell() and s1_channel->branching_count()==0) //auxiliary channel
106  {
107  s1_channel->set_m_max(stot_channel->m()-s2_channel->m());
108  if(!s1_channel->generate_s())
109  {
110  sweight=(value_type)0;
111  return false;
112  }
113  s1_channel->set_status_s_generated();
114  sweight*=s1_channel->s_weight();
115  }
116  if(!s2_channel->on_shell() and s2_channel->branching_count()==0) //auxiliary channel
117  {
118  s2_channel->set_m_max(stot_channel->m()-s1_channel->m());
119  if(!s2_channel->generate_s())
120  {
121  sweight=(value_type)0;
122  return false;
123  }
124  s2_channel->set_status_s_generated();
125  sweight*=s2_channel->s_weight();
126  }
127  }
128  else
129  {
130  if(!ssgen->generate())
131  {
132  sweight=(value_type)0;
133  return false;
134  }
135  s1_channel->set_status_s_generated();
136  s2_channel->set_status_s_generated();
137  sweight=ssgen->weight();
138  }
139  return true;

142  /* Generates the spacelike invariant: */
143 
144  bool generate_t()
145  {
146  value_type sa=this->s_in();
147  value_type sb=sb_channel->s();
148  value_type s1=s1_channel->s();
149  value_type s2=s2_channel->s();
150  value_type s=stot_channel->s();
151 
152  value_type A=0.5*(sa+sb+s1+s2-s-(sa-sb)*(s1-s2)/s);
153  value_type l12=Kallen(s,s1,s2);
154  if(l12<(value_type)0)
155  {
156  tweight=(value_type)0;
157  return false;
158  }
159  value_type B=0.5*std::sqrt(Kallen(s,sa,sb)*l12)/s;
160  value_type tmin=A-B;
161  value_type tmax=A+B;
162 
163  if(!(t_channel->set_s_range(tmin,tmax)))
164  {
165  tweight=(value_type)0;
166  return false;
167  }
168  if(!t_channel->generate_s())
169  {
170  tweight=(value_type)0;
171  return false;
172  }
173  t_channel->set_status_s_generated();
174  tweight=t_channel->s_weight();
175  return (tweight>(value_type)0);
176  }

178  /* Generates momenta with current invariant masses: */
179 
180  bool generate_p()
181  {
182  value_type s1=s1_channel->s();
183  value_type s2=s2_channel->s();
184  value_type s=stot_channel->s();
185 
186  value_type m=stot_channel->m();
187  momentum_type pahat=this->p_in();
188  momentum_type Ptot=pahat+sb_channel->p();
189 
190  boost_to_restframe(pahat,Ptot,m);
191  sphere_generator.generate();
192  value_type pdots(0);
193  for(size_type i=0;i<model_t::dimension-1;++i)
194  {
195  pdots+=(pahat[i+1]*sphere_generator.object()[i]);
196  }
197  value_type pahatabs=std::sqrt(spacetime_type::space_dot(pahat,pahat));
198  value_type k12=Kallen(s,s1,s2);
199  if(k12<(value_type)0)
200  {
201  this->weight()=(value_type)0;
202  return false;
203  }
204  value_type l12=std::sqrt(k12);
205  value_type qhatabs=(value_type)0.5*l12/m;
206 
207  value_type A=0.5*(t_channel->s_max()+t_channel->s_min());
208  value_type B=0.5*(t_channel->s_max()-t_channel->s_min());
209  value_type cos_theta=(t_channel->s()-A)/B;
210  value_type sin_theta=std::sqrt((1-cos_theta)*(1+cos_theta));
211 
212  value_type c1,c2;
213  if(equals(pdots,(value_type)0))
214  {
215  c1=qhatabs*cos_theta/pahatabs;
216  c2=qhatabs*sin_theta;
217  }
218  else
219  {
220  value_type alpha=pahatabs/pdots;
221  value_type y=sin_theta/std::sqrt((alpha-(value_type)1)*(alpha+(value_type)1));
222  c1=qhatabs*(cos_theta+y)/pahatabs;
223  c2=qhatabs*alpha*y;
224  }
225  (s1_channel->p())[0]=(value_type)0.5*(s+s1-s2)/m;
226 
227 
228  for(size_type i=1;i<model_t::dimension;++i)
229  {
230  (s1_channel->p())[i]=c1*pahat[i]-c2*sphere_generator.object()[i-1];
231  }
232  boost_from_restframe(s1_channel->p(),Ptot,m);
233  s1_channel->set_status_p_generated();
234  s1_channel->evaluate_s();
235  s2_channel->p()=Ptot-s1_channel->p();
236  s2_channel->set_status_p_generated();
237  t_channel->p()=this->p_in()-s1_channel->p();
238  t_channel->set_status_p_generated();
239 
240  this->weight()=massless_ps<value_type,2,model_t::dimension>::volume(m)*std::pow(l12/s,int(model_t::dimension-4))*sweight*tweight/((value_type)2*m*pahatabs);
241  return (this->weight()>(value_type)0);
242  }

283  value_type sa=this->s_in();
