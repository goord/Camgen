52  chain_interface(generator_type* gen_,interface_output<model_t>* output_,interface_engine<model_t>* engine_=NULL):base_type(gen_),output(output_),engine(engine_)
53  {
54  std::string namebase(output->file_name);
55  if(namebase.size()==0)
56  {
57  namebase="proc";
58  }
59  namebase.append("_");
60  int digits=0;
61  int step=1;
62  while(step<this->processes())
63  {
64  ++digits;
65  step*=10;
66  }
67  if(digits==0)
68  {
69  digits=1;
70  }
71  for(size_type i=0;i<this->processes();++i)
72  {
73  size_type n=this->gen->process_id(i);
74  std::stringstream ss;
75  ss<<namebase<<std::setw(digits)<<std::setfill('0')<<n;
76  std::string fname(ss.str());
77  interface_output<model_t>* sub_output=output->create(fname);
78  sub_output->open_file();
79  interface_engine<model_t>* sub_engine=NULL;
80  if(engine!=NULL)
81  {
82  sub_engine=engine->clone();
83  sub_engine->set_generator(gen_);
84  sub_engine->add_branches(sub_output);
85  }
86  sub_output->branch(&(this->w),"weight");
87  sub_output->branch(&(this->proc_id),"proc_id");
88  sub_interface sub_int={sub_output,sub_engine};
89  sub_interfaces[n]=sub_int;
90  }
91  }
