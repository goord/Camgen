157  value_type max_weight(const value_type& epsilon) const
158  {
159  if(!(epsilon>(value_type)0))
160  {
161  return maxw;
162  }
163  if(!(epsilon<(value_type)1))
164  {
165  return 0;
166  }
167  typename std::vector<value_type>::const_reverse_iterator it=freqs.rbegin();
168  size_type i=freqs.size()-1;
169  value_type bw(bin_width()),bound(epsilon*wsum/bw),Splus((*it)*i);
170  while(Splus<bound and it!=freqs.rend())

175  }

232  value_type integrate(value_type range,iterator end)
233  {
234  if(pos==end)
235  {
236  return 0;
237  }
238  if(range+offset<1)
239  {
240  offset+=range;
241  return range*(*pos);
242  }
243  value_type r=range+offset-1;
244  value_type result=(1-offset)*(*pos);
245  size_type n=static_cast<size_type>(std::floor(r));
246  for(size_type i=0;i<n;++i)
247  {
248  if((++pos)==end)
249  {
250  break;
251  }
252  result+=(*pos);
253  }
254  if(pos!=end)
255  {
256  offset=r-n;

271  /* Rebins such that the new maximal weight w is contained in the
272  * last bin and the number of bins is not changed. */
273 
274  void rebin(const value_type& w)
275  {
276  if(w<maxw)
277  {
278  return;
279  }
280  value_type oldwidth=bin_width();
281  value_type newwidth=w/(freqs.size()-1);
282  std::vector<value_type>newfreqs(freqs.size());
283  bin_iterator it(freqs.begin());
284  for(size_type i=0;i<newfreqs.size()-1;++i)
285  {
286  newfreqs[i]=it.integrate(newwidth/oldwidth,freqs.end());
287  }
288  value_type lastfill=it.integrate(newwidth/oldwidth,freqs.end());
289  if(lastfill>(value_type)0)
290  {
291  log(log_level::warning)<<CAMGEN_STREAMLOC<<"the last bin is incorrectly filled with "<<lastfill<<endlog;
292  }
293  freqs=newfreqs;
294  }
