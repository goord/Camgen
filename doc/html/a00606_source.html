73  bool refresh_params()
74  {
75  mm=(m==NULL)?(value_type)0:((*m)*(*m));
76  mw=(m==NULL)?(value_type)0:((w==NULL)?(value_type)0:((*m)*(*w)));
77  if(mw==(value_type)0)
78  {
79  log(log_level::warning)<<CAMGEN_STREAMLOC<<"invalid Breit-Wigner parameters mass: "<<mass()<<" GeV, width: "<<width()<<" GeV encountered"<<endlog;
80  return false;
81  }
82  this->max_cumulant_lim=pi2/mw;
83  this->min_cumulant_lim=-this->max_cumulant_lim;
84  mw2=mw*mw;
85  alpha=std::sqrt(mm*mm+mw2);
86  xi_plus=std::sqrt((value_type)2*(alpha+mm));
87  xi_min=std::sqrt((value_type)2*(alpha-mm));
88  nu=std::sqrt(std::complex<value_type>(mm,-mw));
89  nubar=std::sqrt(std::complex<value_type>(mm,mw));
90  return (this->refresh_lower_bound() and this->refresh_upper_bound());
91  }
