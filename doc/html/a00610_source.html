19 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
20  * Definition of the CM_algorithm class template. The CM_algorithm constitutes *
21  * Camgen's user interface for computing process amplitudes. It creates a list *
22  * of process trees, which can be selected via the set_process member function, *
23  * and then evaluated by calling evaluate(). It allows access to the external *
24  * particle's phase space by calling get_phase_space(), swapping degrees of *
25  * freedom, counting diagrams and memory usage etc. *
26  * *
27  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */

194  process_iterator set_process(const std::string& proc)
195  {
196  vector<std::string,N_in+N_out>namevec;
197  if(!decompose_process<N_in,N_out>(proc,namevec))
198  {
199  process_it=processes.end();
200  tree_it=trees.end();
201  return process_it;
202  }
203  flavour_vector fv;
204  const particle<model_t>* phi;
205  for(size_type i=0;i<N_in+N_out;++i)
206  {
207  phi=model_wrapper<model_t>::get_particle(namevec[i]);
208  if(phi==NULL)
209  {
210  log(log_level::warning)<<CAMGEN_STREAMLOC<<"failed to set process "<<proc<<": particle "<<namevec[i]<<" not found in model"<<endlog;
211  process_it=processes.end();
212  tree_it=trees.end();
213  return process_it;
214  }
215  else
216  {
217  fv[i]=phi->get_flavour();
218  }
219  }
220  return set_process_flavours(fv);
221  }

228  process_iterator set_process_flavours(const flavour_vector& fv)
229  {
230  /* Construct a sorted copy of the argument: */
231 
232  flavour_vector sorted_fv(fv);
233  std::sort(sorted_fv.begin(),sorted_fv.begin()+N_in);
234  std::sort(sorted_fv.begin()+N_in,sorted_fv.end());
235 
236  /* Look for the sorted flavours in the process list: */
237 
238  if(sorted_by_flavour)
239  {
240  process_it=std::lower_bound(processes.begin(),processes.end(),sorted_fv,flav_vec_comp<model_t,N_in,N_out>);
241  }
242  else
243  {
244  process_it=processes.begin();
245  for(;process_it!=processes.end();++process_it)
246  {
247  if(!(process_it->get_sorted_flavours()<sorted_fv)){break;}
248  }
249  }
250 
251  /* Reset the ordering vector to 'no ordering' mode: */
252 
253  reset_ordering();
254 
255  /* If the lower bound search ends within the tree, there are two
256  scenario's: */
257 
258  if(process_it!=processes.end())
259  {
260  /* If the process was found, set the ordering to the correct
261  * values and move the tree iterator to the corresponding
262  * tree: */
263 
264  tree_it=process_it->get_tree();
265  if(process_it->get_sorted_flavours()==sorted_fv)
266  {
267  permutation<flavour_vector>perm(fv,process_it->get_flavours());
268  perm(ordering);
269  return process_it;
270  }
271 
272  /* Else, attempt to insert the process in the list: */
273 
274  process_it=process<model_t,N_in,N_out>::insert_process_by_flavour(processes,process_it,fv);
275 
276  /* If the insertion was succesful, insert a corresponding
277  * tree: */
278 
279  if(process_it!=processes.end())
280  {
281  tree_it=process_it->insert_tree(trees,tree_it);
282  tree_it->build();
283  tree_it->clean();
284  tree_it->set_Fermi_signs();
285  tree_it->initialise_currents();
286  tree_it->assign_momenta();
287  return process_it;
288  }
289 
290  /* If the insertion was not succesful, set the iterators to
291  * the end of the lists: */
292 
293  tree_it=trees.end();
294  return process_it;
295  }
296 
297  /* If the search result outside the process list, attempt to
298  * insert the process at the back: */
299 
300  process_it=process<model_t,N_in,N_out>::insert_process_by_flavour(processes,process_it,fv);
301 
302  /* If the insertion was succesful, insert the corresponding tree
303  * at the back of the tree list: */
304 
305  if(process_it != processes.end())
306  {
307  tree_it=process_it->insert_tree(trees,trees.end());
308  tree_it->build();
309  tree_it->clean();
310  tree_it->set_Fermi_signs();
311  tree_it->initialise_currents();
312  tree_it->assign_momenta();
313  }
314 
315  /* Else, set the tree iterator outside the list: */
316 
317  else
318  {
319  tree_it=trees.end();
320  }
321  return process_it;
322  }
