78  bool generate_fs()
79  {
80  momentum_type P=this->P_in();
81  value_type stot=spacetime_type::dot(P,P);
82  if(stot<(value_type)0)
83  {
84  log(log_level::warning)<<CAMGEN_STREAMLOC<<"negative total invariant mass-squared encountered: P = "<<P<<endlog;
85  this->fs_weight()=(value_type)0;
86  return false;
87  }
88  value_type Etot=std::sqrt(stot);
89  if(Etot<this->ps_generator_base<model_type>::M_out_sum())
90  {
91  log(log_level::warning)<<CAMGEN_STREAMLOC<<"total energy "<<Etot<<" too small to create final state with M = "<<this->ps_generator_base<model_type>::M_out_sum()<<endlog;
92  this->fs_weight()=(value_type)0;
93  return false;
94  }
95 
96  /* CM-frame massless momenta generation: */
97 
98  momentum_type k;
99  k.assign((value_type)0);
100  for(size_type i=0;i<N_out;++i)
101  {
102  value_type rho=rn_stream::throw_number();
103  for(size_type n=0;n<D-3;++n)
104  {
105  rho*=rn_stream::throw_number();
106  }
107  phat=-std::log(rho);
108  (this->p_out(i))[0]=phat;
109  spheregen->generate();
110  for(size_type mu=1;mu<D;++mu)
111  {
112  this->p_out(i)[mu]=spheregen->object()[mu-1];
113  }
114  k+=(this->p_out(i));
115  }
116 
117  /* Rescaling: */
118 
119  value_type ksq=spacetime_type::dot(k,k);
120  value_type scale_factor=Etot/ksq;
121  value_type u=std::sqrt(std::abs(ksq));
122 
123  value_type kq,w;
124  for(size_type i=0;i<N_out;++i)
125  {
126  kq=spacetime_type::dot(k,this->p_out(i));
127  value_type& e=this->p_out(i)[0];
128  w=(kq+u*e)/(u+k[0]);
129  e=scale_factor*kq;
130  for(size_type mu=1;mu<D;++mu)
131  {
132  this->p_out(i)[mu]=scale_factor*(u*(this->p_out(i)[mu])-w*k[mu]);
133  }
134  }
135  if(!this->massive_fs())
136  {
137  this->fs_weight()=massless_ps<value_type,N_out,D>::volume(Etot);
138  }
139  else
140  {
141  /* Massive case, with extra constraction: */
142 
143  value_type xi=(value_type)1;
144  value_type f,df,d;
145  vector<value_type,N_out>Esq;
146  for(size_type i=0;i<N_out;++i)
147  {
148  Esq[i]=(this->p_out(i)[0])*(this->p_out(i)[0]);
149  }
150  for(size_type i=0;i<NR_iterations();++i)
151  {
152  f=-Etot;
153  df=(value_type)0;
154  for(size_type j=0;j<N_out;++j)
155  {
156  d=std::sqrt(xi*xi*Esq[j]+this->M2_out(j));
157  f+=d;
158  df+=Esq[j]/d;
159  }
160  xi-=f/(xi*df);
161  }
162  for(size_type i=0;i<N_out;++i)
163  {
164  this->p_out(i)[0]=std::sqrt(xi*xi*Esq[i]+this->M2_out(i));
165  for(size_type mu=1;mu<D;++mu)
166  {
167  this->p_out(i)[mu]*=xi;
168  }
169  }
170 
171  /* Extra weight due to phase space contraction: */
172 
173  (this->fs_weight())=massless_ps<value_type,N_out,D>::volume(Etot)*std::pow(xi,int((D-2)*N_out-D+1))*Etot;
174  value_type factor=(value_type)1;
175  value_type denom=(value_type)0;
176  for(size_type i=0;i<N_out;++i)
177  {
178  value_type e=this->p_out(i)[0];
179  value_type psq=e*e-this->M2_out(i);
180  factor*=(std::sqrt(psq)/e);
181  denom+=(psq/e);
182  }
183  (this->fs_weight())*=(factor/denom);
184  }
185 
186  /* Boosting CM-momenta to the lab frame: */
187 
188  for(size_type i=0;i<N_out;++i)
189  {
190  boost_from_restframe(this->p_out(i),P,Etot);
191  }
192 
193  return true;
194  }

196  /* Evaluates the weight for the given final state. */

198  bool evaluate_fs_weight()
199  {
200  value_type stot=this->s_hat();
201  if(stot<(value_type)0)
202  {
203  log(log_level::warning)<<CAMGEN_STREAMLOC<<"negative total invariant mass-squared encountered"<<endlog;
204  this->fs_weight()=(value_type)0;
205  return false;
206  }
207  value_type Etot=std::sqrt(stot);
208  if(Etot<this->ps_generator_base<model_type>::M_out_sum())
209  {
210  log(log_level::warning)<<CAMGEN_STREAMLOC<<"total energy too small to create final state"<<endlog;
211  this->fs_weight()=(value_type)0;
212  return false;
213  }
214  this->fs_weight()=massless_ps<value_type,N_out,D>::volume(Etot);
215  if(this->massive_fs())
216  {
217  value_type factor=(value_type)1;
218  value_type denom=(value_type)0;
219  value_type num=(value_type)0;
220  for(size_type i=0;i<N_out;++i)
221  {
222  value_type E=this->p(i)[0];
223  value_type pvec=std::sqrt(E*E-this->M2_out(i));
224  factor*=(pvec/E);
225  num+=pvec;
226  denom+=(pvec*pvec/E);
227  }
228  this->fs_weight()*=(Etot*std::pow(num/Etot,int((D-2)*N_out-D+1))*factor/denom);
229  }
230  return true;
231  }
