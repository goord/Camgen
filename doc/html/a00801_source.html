125  /* Recursion relation evaluating the tensor subamplitude: */
126 
127  static void first(ARG_LIST)
128  {
129  value_type c;
130  for(size_type mu=0;mu<vector_size;++mu)
131  {
132  for(size_type nu=0;nu<=mu;++nu)
133  {
134  c=factor*C_0*(A_1[mu]*A_2[nu]+A_2[mu]*A_1[nu]);
135  A_0[mu*vector_size+nu]+=c;
136  A_0[nu*vector_size+mu]+=c;
137  }
138  }
