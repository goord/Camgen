194  /* Generation operator implementation. Generates complex phases for
195  * the positive helicity coefficients, the corresponding negative
196  * helicity state receive the complex conjugate phase. If a
197  * zero-helicity state exists, it gets coefficient 1. */
198 
199  bool generate()
200  {
201  for(size_type i=0;i<N_in+N_out;++i)
202  {
203  int j=this->maximal_helicity(i);
204  for(int k=1;k<j+1;++k)
205  {
206  c_value_type c=std::polar((value_type)1,rn_stream::throw_number(twopi));
207  this->helicity_phase(i,k)=c;
208  this->helicity_phase(i,-k)=std::conj(c);
209  }
