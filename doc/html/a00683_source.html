29  template<class value_t,std::size_t D>vector<value_t,D>& boost_from_restframe(vector<value_t,D>& q,const vector<value_t,D>& p,const value_t& sqrts)
30  {
31  typedef typename Minkowski_type::template implementation<value_t,D> spacetime_type;
32 
33  value_t pdotq=spacetime_type::space_dot(p,q);
34  value_t y=(q[0]+pdotq/(p[0]+sqrts))/sqrts;

36  CAMGEN_ERROR_IF((sqrts==(value_t)0),"attempt to boost vector to a light-like frame");
37
