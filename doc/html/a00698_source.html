98  generator_type* select_channel()
99  {
100  if(channels.size()==0)
101  {
102  current_channel_iterator=channels.end();
103  return NULL;
104  }
105  current_channel_iterator=channels.begin();
106  if(channels.size()==1)
107  {
108  return current_channel_iterator->generator;
109  }
110  value_type rho=rn_stream::throw_number();
111  value_type r(0);
112  while(r<=rho and current_channel_iterator!=channels.end())
113  {
114  r+=(current_channel_iterator->alpha);
115  ++current_channel_iterator;
116  }
117  --current_channel_iterator;
118  if(r>(value_type)1+numeric_configuration<value_type>::epsilon_abs)
119  {
120  log(log_level::warning)<<CAMGEN_STREAMLOC<<"multichannel weights incorrectly normalized."<<r<<endlog;
121  }
122  return current_channel();
123  }

172  bool evaluate_weight(bool recurse)
173  {
174  if(channels.size()==0)
175  {
176  this->weight()==(value_type)1;
177  return true;
178  }
179  value_type g(0);
180  bool zero_weight=false;
181  for(channel_iterator it=channels.begin();it!=channels.end();++it)
182  {
183  if(recurse)
184  {
185  if(!it->generator->evaluate_weight())
186  {
187  this->weight()=(value_type)0;
188  return false;
189  }
190  }
191  if(it->generator->weight()==(value_type)0 and it->alpha>(value_type)0)
192  {
193  zero_weight=true;
194  }
195  else if(it->generator->weight()!=(value_type)0)
196  {
197  g+=(it->alpha/it->generator->weight());
198  }
199  }
200  this->weight()=zero_weight?(value_type)0:((value_type)1)/g;
201  return true;
202  }

331  {

336  value_type norm(0);
