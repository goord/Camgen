49  static void initialise(std::size_t NR_iterations=10)
50  {
51  if(!initialised)
52  {
53  value_type pi=std::acos(-(value_type)1);
54  for(std::size_t n=0;n<N;++n)
55  {
56  roots[n]=std::cos(value_type(4*n+3)*pi/value_type(4*N+2));
57  for(std::size_t i=0;i<NR_iterations;++i)
58  {
59  roots[n]-=value(roots[n])/derivative(roots[n]);
60  }
61  value_type p=derivative(roots[n]);
62  weights[n]=(value_type)2/(((value_type)1-roots[n]*roots[n])*p*p);
63  }

101  template<value_type f(const value_type&)>static value_type Gauss_integrate(const value_type& a,const value_type& b)
102  {
103  initialise();
104  value_type s=(value_type)0.5*(a+b);
105  value_type d=(value_type)0.5*(b-a);
106  value_type x;
107  value_type result=0;
108  for(unsigned i=0;i<N;++i)

112  }
