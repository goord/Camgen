116  {

122  value_type rho=rn_stream::throw_number(smin,smax);
123  reg_bin=grid->root_bin->find_weight(rho);

125  if(!q)
126  {

128  return false;
129  }
130  this->weight()=(smax-smin)*reg_bin->weight();

133  return true;
