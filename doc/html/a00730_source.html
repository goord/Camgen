503  void generate_unweighted(bool verbose=false)
504  {
505  if(verbose)
506  {
507  std::cout<<"generating event for ";
508  print_process(std::cout);
509  std::cout<<"......";

514  generate();

530  this->weight()=tot_weight;

552  bool evaluate_weight()
553  {
554  if(!(evaluate_phase_space_weight() and evaluate_helicity_weight() and evaluate_colour_weight()))
555  {
556  this->weight()=(value_type)0;
557  set_integrands(0);
558  return false;
559  }
560  this->weight()=ps_weight*hel_weight*col_weight;
561  value_type f=pb_conversion*symmetry_factor*ps_factor*hel_factor*col_factor;
562  if(f!=(value_type)0)
563  {
564  evaluate_amplitude();
565  if(!accept(me))
566  {
567  this->weight()=(value_type)0;
568  set_integrands(0);
569  return false;
570  }
571  f*=me;
572  }
573  set_integrands(f);
574  if(auto_update)
575  {
576  update();
577  this->refresh_cross_section();
578  }
579  if(update_counter==0)
580  {
581  return true;
